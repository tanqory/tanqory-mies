```jsx title="Accordion Component"
function AccordionExample() {
  const [openItems, setOpenItems] = useState({});
  
  const toggleItem = (itemKey) => {
    setOpenItems(prev => ({
      ...prev,
      [itemKey]: !prev[itemKey]
    }));
  };
  
  const accordionData = [
    { key: 'first', title: 'First Accordion', content: 'First Content' },
    { key: 'second', title: 'Second Accordion', content: 'Second Content' },
    { key: 'third', title: 'Third Accordion', content: 'Third Content' }
  ];
  
  return (
    <div style={{ maxWidth: '600px', margin: '0 auto' }}>
      {accordionData.map(item => (
        <div key={item.key} style={{ 
          border: '1px solid #ddd', 
          marginBottom: '8px',
          borderRadius: '4px'
        }}>
          <div 
            onClick={() => toggleItem(item.key)}
            style={{
              padding: '16px',
              background: '#f8f9fa',
              cursor: 'pointer',
              display: 'flex',
              justifyContent: 'space-between',
              alignItems: 'center'
            }}
          >
            <span>{item.title}</span>
            <span>{openItems[item.key] ? '▼' : '▶'}</span>
          </div>
          {openItems[item.key] && (
            <div style={{ padding: '16px', background: 'white' }}>
              {item.content}
            </div>
          )}
        </div>
      ))}
    </div>
  );
}
```
